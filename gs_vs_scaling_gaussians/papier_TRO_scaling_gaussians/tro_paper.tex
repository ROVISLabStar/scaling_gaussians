\section{Scale Inflation: Principle}
\label{sec:principle}

\begin{figure*}[t]
    \centering
    \includegraphics[width=\textwidth]{alpha_illustration.pdf}
    \caption{Effect of the scale factor $\alpha$ on 3D Gaussian primitives and their splatting. Top: three ellipsoidal Gaussians in 3D space, inflated by $\alpha$. Bottom: corresponding rendered (splatted) images. At $\alpha=1.0$, the three Gaussians produce sharp, distinct blobs. As $\alpha$ increases, the Gaussians grow and overlap, producing a progressively smoother image --- analogous to the ``power of attraction'' of Crombez et al.~\cite{crombez2019tro}, but operating directly in 3D scene space.}
    \label{fig:alpha_illustration}
\end{figure*}

\begin{figure*}[t]

    \caption{Influence of the scale factor $\alpha$ on 3DGS rendering for two scenes: \textit{room} (top) and \textit{garden} (bottom). Increasing $\alpha$ inflates the 3D Gaussian covariances, producing progressively smoother rendered images. The outdoor scene (garden) preserves structure at higher $\alpha$ due to its larger depth range, which explains why it requires higher $\alpha$ for optimal convergence (cf. \subsection{Cost Function Landscape}

\begin{figure*}[t]
    \centering
    \includegraphics[width=\textwidth]{cost_landscape_2d.pdf}
    \caption{Cost function landscape $\mathcal{C}(\Delta t_x, \Delta t_y)$ for different scale factors $\alpha$. The cost is computed as the SSD between the desired image and images rendered at perturbed poses along two translation axes. Increasing $\alpha$ smooths the cost function, suppressing local minima and widening the convergence basin. Beyond $\alpha \approx 2.4$, the gradient vanishes, hindering convergence.}
    \label{fig:cost_landscape}
\end{figure*}

\subsection{Robustness to 3D Reconstruction Quality}

Unlike classical VS where noise comes from the camera sensor, in 3DGS-based VS the dominant source of imprecision is the quality of the 3D reconstruction itself. We evaluate robustness by perturbing the Gaussian parameters directly: position ($\sigma_p$ as multiples of mean Gaussian scale $\bar{s}$), scale ($\sigma_s$ as relative std), and SH color coefficients ($\sigma_c$).

\begin{figure*}[t]
    \centering
    \includegraphics[width=\textwidth]{3d_noise_renders.pdf}
    \caption{Effect of 3D Gaussian perturbations on rendered images. Position and color noise produce mild degradation. Scale noise at $\sigma_s=1.0$ significantly distorts the rendering.}
    \label{fig:3d_noise_renders}
\end{figure*}

\begin{figure*}[t]
    \centering
    \includegraphics[width=\textwidth]{3d_noise_robustness.pdf}
    \caption{Robustness to 3D Gaussian parameter perturbations (pair $5 \to 10$). Left: position noise. Center: scale noise. Right: SH color noise. Both methods are robust to position and color perturbations. At extreme scale noise ($\sigma_s=1.0$), the inflated method ($\alpha=2.0$) fails ($\times$) because inflation amplifies the noisy scales, while the original remains robust.}
    \label{fig:3d_noise}
\end{figure*}
